
# Agregamos todo (archivos y directorios) al repositorio
git add .

# O agregamos un archivo al repositorio
git add 'nombre del archivo o carpeta'

# Ver archivos añadidos y no añadidos
git status

# Descartar cambios agregados sin borrarlos
git reset .

# Descartar cambios agregados o no agregados y borrarlos
git restore .
